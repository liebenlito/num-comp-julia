Kekurangan interpolasi Lagrange¶

Terdapat beberapa kekurangan interpolasi Lagrange, antara lain:

  • Lambat untuk dihitung, setiap li(x)l_i(x)li​(x) berbeda,
  • Tidak fleksibel: jika kita ganti/tambahkan titik data baru, kita harus menghitung ulang semua lil_ili​ nya

function Lagrange_basis(x_data)
    n = length(x_data)
    L = zeros(n, n)
    for i in 1:n
        for j in 1:n
            if i == j
                L[i, j] = 1.0
            else 
                L[i, j] = 0.0
            end
        end
    end
    return L
end

function poly_Lagrange(x, x_data, koef)
    n = length(x_data)
    hasil = 0.0
    for i in 1:n
        term = koef[i]
        for j in 1:n
            if i != j
                term *= (x - x_data[j]) / (x_data[i] - x_data[j])
            end
        end
        hasil += term
    end
    return hasil
end
