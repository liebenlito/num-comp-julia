TOL = 1e-5
M = 1000

POWER = y
TERM = y

while N <= M
    SIGN = -SIGN
    SUM = SUM + SIGN * TERM
    POWER = POWER * y
    TERM = POWER / (N + 1)
    
    if abs(TERM) < TOL
